def create
  self.resource = warden.authenticate!(auth_options)
  sign_in(resource_name, resource)
  yield resource if block_given?
  
  render json: {
    message: 'Logged in successfully',
    user: {
      id: resource.id,
      email: resource.email
    }
  }, status: :ok
end

def destroy
